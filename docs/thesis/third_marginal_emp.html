46  🚧 Empirical Study w/ Marginals

We estimate the intensity surface via KDE, then use simple summaries of the estimated intensities as features.

Global settings

# Main parameter grids
lambda_vals <- c(100, 500)
phi_vals <- c(1, 2)

# Fixed parameters unless otherwise varied

# Number of simulated datasets per setting
n_sims <- 100

# Grid of intensity values for empirical CDF / survival curves
intensity_grid <- seq(0, 1000, length.out = 5000)

# Thresholds for tail summaries, expressed as multiples of lambda
tail_multiples <- c(1.5, 2, 3, 5)

# Kernel smoothing bandwidth for estimated intensity
sigma_kernel <- 0.03

Helpers:

Code 
# Sim + extraction
simulate_pattern <- function(family, W, lambda, phi, s, df = 1) {
  family <- match.arg(family, c("LGCP", "CSCP"))
  
  out <- switch(
    family,
    LGCP = sim_lgcp(W, lambda = lambda, phi = phi, s = s),
    CSCP = sim_cscp(W, lambda = lambda, phi = phi, s = s, df = df)
  )
  
  out$pp
}

# Estimated intensity surface
estimate_intensity_surface <- function(X, sigma = 0.03, npix = 128) {
  spatstat.options(npixel = npix)
  
  density.ppp(
    X,
    sigma = sigma,
    at = "pixels",
    edge = TRUE,
    positive = TRUE
  )
}

extract_im_values <- function(im_obj) {
  vals <- as.vector(im_obj$v)
  vals[is.finite(vals)]
}

# Empirical CDF
ecdf_on_grid <- function(values, grid) {
  Fhat <- ecdf(values)
  Fhat(grid)
}

# Summary helper
summarise_intensity_values <- function(values, lambda, tail_multiples = c(1.5, 2, 3, 5)) {
  out <- tibble(
    mean_est = mean(values),
    sd_est = sd(values),
    q50 = unname(quantile(values, 0.50)),
    q90 = unname(quantile(values, 0.90)),
    q95 = unname(quantile(values, 0.95)),
    q99 = unname(quantile(values, 0.99))
  )
  
  tail_probs <- map_dfc(tail_multiples, function(mult) {
    tibble(!!paste0("p_gt_", mult, "x_mean") := mean(values > mult * lambda))
  })
  
  bind_cols(out, tail_probs)
}

Replicate function

One replicate returns: - estimated intensity values - empirical CDF / survival on a common grid - summary statistics

Code 
one_empirical_marginal_rep <- function(family, rep, W, lambda, phi, s, df,
                                       sigma_kernel, npix,
                                       intensity_grid,
                                       tail_multiples) {
  X <- simulate_pattern(
    family = family,
    W = W,
    lambda = lambda,
    phi = phi,
    s = s,
    df = df
  )
  
  lambda_hat_im <- estimate_intensity_surface(
    X,
    sigma = sigma_kernel,
    npix = npix
  )
  
  vals <- extract_im_values(lambda_hat_im)
  
  curve_df <- tibble(
    family = family,
    rep = rep,
    lambda = lambda,
    phi = phi,
    s = s,
    x = intensity_grid,
    cdf = ecdf_on_grid(vals, intensity_grid)
  )
  
  summary_df <- summarise_intensity_values(
    values = vals,
    lambda = lambda,
    tail_multiples = tail_multiples
  ) %>%
    mutate(
      family = family,
      rep = rep,
      lambda = lambda,
      phi = phi,
      s = s,
      n_points = X$n
    ) %>%
    relocate(family, rep, lambda, phi, s, n_points)
  
  list(
    curve_df = curve_df,
    summary_df = summary_df,
    values = vals,
    X = X,
    lambda_hat_im = lambda_hat_im
  )
}

46.1 Running the study

Main engine:

Code 
run_empirical_marginal_study <- function(
  W,
  lambda_vals,
  phi_vals,
  s = 0.05,
  df = 1,
  n_sims = 100,
  sigma_kernel = 0.03,
  npix = 128,
  intensity_grid = seq(0, 5000, length.out = 5000),
  tail_multiples = c(1.5, 2, 3, 5)
) {
  settings <- tidyr::crossing(
    lambda = lambda_vals,
    phi = phi_vals
  )
  
  n_settings <- nrow(settings)
  
  results <- pmap(
    list(settings$lambda, settings$phi, seq_len(n_settings)),
    function(lambda, phi, setting_id) {
      
      # cat(
      #   "\n====================================================\n",
      #   "Setting ", setting_id, " of ", n_settings,
      #   " | lambda = ", lambda,
      #   " | phi = ", phi, "\n",
      #   "====================================================\n",
      #   sep = ""
      # )
      
      reps <- map(
        seq_len(n_sims),
        function(rep) {
          # cat("  Rep", rep, "of", n_sims, "\n")
          
          list(
            LGCP = one_empirical_marginal_rep(
              family = "LGCP",
              rep = rep,
              W = W,
              lambda = lambda,
              phi = phi,
              s = s,
              df = df,
              sigma_kernel = sigma_kernel,
              npix = npix,
              intensity_grid = intensity_grid,
              tail_multiples = tail_multiples
            ),
            CSCP = one_empirical_marginal_rep(
              family = "CSCP",
              rep = rep,
              W = W,
              lambda = lambda,
              phi = phi,
              s = s,
              df = df,
              sigma_kernel = sigma_kernel,
              npix = npix,
              intensity_grid = intensity_grid,
              tail_multiples = tail_multiples
            )
          )
        }
      )
      
      curve_df <- bind_rows(
        map(reps, ~ .x$LGCP$curve_df),
        map(reps, ~ .x$CSCP$curve_df)
      )
      
      summary_df <- bind_rows(
        map(reps, ~ .x$LGCP$summary_df),
        map(reps, ~ .x$CSCP$summary_df)
      )
      
      list(
        curve_df = curve_df,
        summary_df = summary_df,
        raw_reps = reps
      )
    }
  )
  
  list(
    curve_df = bind_rows(map(results, "curve_df")),
    summary_df = bind_rows(map(results, "summary_df")),
    raw = results
  )
}

Note

So essentially: We have simulated a bunch of replicates with different lambda / phi and run KDE over them all. And now we have looked at the “empirical CDF” by just essentially making a histogram of pixel intensity values.

And now we are saying, look, there is a difference as lambda get large, but now we want to know is this difference:

  1. Larger than the random variation from estimating? and

  2. Large enough for us to make a practically useable classifier?
